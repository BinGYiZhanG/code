### 思路:
* 可能在要划线之前,就是'+'号情况
* 可能在划线的中间存在交叉,所以不能只在开头和结尾判断交叉


```
#include <bits/stdc++.h>

using namespace std;

struct Node{
    int x;
    int y;
};
int m,n,q;
char mp[110][110];
int dx[4]={1,-1,0,0};
int dy[4]={0,0,1,-1};
int vis[110][110];

void fill_char(int x,int y,char c){
    queue<Node> que;
    Node nd,tmp;
    memset(vis,false,sizeof(vis));
    nd.x=n-1-y,nd.y=x;
    mp[n-1-y][x]=c;
    vis[n-1-y][x]=true;
    que.push(nd);
    while(!que.empty()){
        nd=que.front();
        que.pop();

        for(int i=0;i<4;i++){
            tmp.x=nd.x+dx[i];
            tmp.y=nd.y+dy[i];
            if(vis[tmp.x][tmp.y]||mp[tmp.x][tmp.y]=='|'||mp[tmp.x][tmp.y]=='-'||mp[tmp.x][tmp.y]=='+'
               ||tmp.x<0||tmp.x>=n||tmp.y<0||tmp.y>=m)
                continue;
            //printf("%d %d\n",tmp.x,tmp.y);
            vis[tmp.x][tmp.y]=true;
            mp[tmp.x][tmp.y]=c;
            que.push(tmp);
        }
    }
}

void paint_line(int x1,int y1,int x2,int y2){
    if(x1>x2||y1>y2){///让左端点是小的那个,右端点是大的那个
        swap(x1,x2);
        swap(y1,y2);
    }
    if(y1==y2){///列相同==>实际上的行相同
        int j=x1;
        if(mp[n-1-y1][j]=='|'||mp[n-1-y1][j]=='+'){///处理开始的交叉点
            mp[n-1-y1][j]='+';
            j++;
        }
        for(;j<=x2;j++){
            //if(j==x2&&mp[n-1-y1][j]=='|')///处理最后的交叉点
            //    mp[n-1-y1][j]='+';
            //else if(j==x2&&mp[n-1-y1][j]=='+')///跳过最后也是+的位置
            //    ;
            if(mp[n-1-y1][j]=='+'||mp[n-1-y1][j]=='|')
                mp[n-1-y1][j]='+';
            else
                mp[n-1-y1][j]='-';
        }
    }
    if(x1==x2){///行相同==>实际上的列相同
        ///
        int i=n-y2-1;
        if(mp[n-y2-1][x1]=='-'||mp[n-y2-1][x1]=='+'){///处理开始的交叉点
            mp[n-y2-1][x1]='+';
            i++;
        }
        for(;i<=n-y1-1;i++){
            //if(i==n-y1-1&&mp[i][x1]=='-')///处理最后的交叉点
              //  mp[i][x1]='+';
            //else if(i==n-y1-1&&mp[i][x1]=='+')
              //  ;
            if(mp[i][x1]=='-'||mp[i][x1]=='+')///不用上述代码的原因,可能在一条线的中间位置存在交叉,
                                                ///所以在划线时得时刻判断是否存在交叉
                mp[i][x1]='+';
            else
                mp[i][x1]='|';
        }

    }
}

int main(){
    scanf("%d%d%d",&m,&n,&q);
    for(int i=0;i<n;i++)///初始化
        for(int j=0;j<m;j++)
            mp[i][j]='.';

    int flag,x,y,x1,y1,x2,y2;
    char c;
    for(int cmd=0;cmd<q;cmd++){
        scanf("%d",&flag);
        if(flag){///填充操作
            cin>>x>>y>>c;
            //printf("%d %d %c\n",x,y,c);
            fill_char(x,y,c);
        }
        else{///画线段操作
            scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
            paint_line(x1,y1,x2,y2);
        }
    }
    for(int i=0;i<n;i++){///初始化
        for(int j=0;j<m;j++){
            printf("%c",mp[i][j]);
        }
        printf("\n");
    }

    return 0;
}
```
