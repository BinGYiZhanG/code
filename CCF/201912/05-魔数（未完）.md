#include <bits/stdc++.h>

using namespace std;

struct Bign{
    int d[1010];
    int len;
    bign(){
        memset(d,0,sizeof(d));
        len=0;
    }
};

Bign change(int num){
    string str;
    while(num!=0){
        str+=(num%10);
        num/=10;
    }
    reverse(str.begin(),str.end());
    Bign a;
    a.len=str.size();
    for(int i=0;i<a.len;i++){
        a.d[i]=str[a.len-1-i]-'0';
    }
    return a;
}

Bign add(Bign a,Bign b){
    Bign c;
    int carry=0;
    for(int i=0;i<a.len||i<b.len;i++){
        int tmp=a.d[i]+b.d[i]+carry;
        c.d[c.len++]=tmp%10;
        carry=tmp/10;
    }
    if(carry)
        c.d[c.len++]=carry;
    return c;
}

Bign sub(Bign a,Bign b){
    Bign c;
    for(int i=0;i<a.len||i<b.len;i++){
        if(a.d[i]<b.d[i]){
            a.d[i+1]--;
            a.d[i]+=10;
        }
        c.d[c.len++]=a.d[i]-b.d[i];
    }
    while(c.len-1>=1&&c.d[c.len-1]==0)
        c.len--;
    return c;
}

Bign multi(Bign a,int b){
    Bign c;
    int carry=0;
    for(int i=0;i<a.len;i++){
        int tmp=a.d[i]*b+carry;
        c.d[c.len++]=tmp%10;
        carry=tmp/10;
    }
    while(carry){
        c.d[c.len++]=carry%10;
        carry/=10;
    }
    return c;
}

Bign divide(bign a,int b,int& r){
    Bign c;
    c.len=a.len;
    for(int i=a.len-1;i>=0;i--){
        r=r*10+a.d[i];
        if(r<b)
            c.d[i]=0;
        else{
            c.d[i]=r/b;
            r=r%b;
        }
    }
    while(c.len-1>=1&&c.d[c.len-1]==0)
        c.len--;
    return c;
}

    for(int i=a.len-1;i>=0;i--)
