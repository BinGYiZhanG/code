### 思路：
* 队列，分配座位
* [转自CSDN](https://blog.csdn.net/qq_38996065/article/details/79793908)

```
#include <bits/stdc++.h>

using namespace std;

struct Queue{
    int lsum;///每队人数
}que[110];
int n,sum=0;

int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&que[i].lsum);
        que[i].lsum*=10;///更新各队人数
        sum+=que[i].lsum;///记录总人数
    }
    queue<int> q[n+1];
    int index=1;///记录座位编号
    while(sum){
        for(int i=1;i<=n;i++){
            if(que[i].lsum!=0){///结束条件
                sum--;
                que[i].lsum--;
                if(que[i].lsum==sum){
                ///如果最后只剩下1所学校的队伍还没有分配座位
                ///需要隔座而坐
                    q[i].push(index);
                    index+=2;
                }
                else{
                ///每队依次进入第1,2,3...号成员
                    q[i].push(index);
                    index+=1;
                }
            }
        }
    }

    int cnt;
    for(int i=1;i<=n;i++){
        printf("#%d\n",i);
        cnt=0;
        while(!q[i].empty()){
            cnt++;
			if(cnt%10==1)///处理中间的空格
                printf("%d",q[i].front());
			else
                printf(" %d",q[i].front());
			if(cnt%10==0)///每10个输出回车
                printf("\n");
			q[i].pop();
        }
    }
    return 0;
}
```
