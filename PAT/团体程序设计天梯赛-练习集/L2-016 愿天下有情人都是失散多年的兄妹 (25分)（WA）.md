* 题解1的dfs
* 需要初始化父母结点，否则结果出错
```
#include <bits/stdc++.h>

using namespace std;

struct Node{
    char sex;
    int fid,mid;
}cp[100000];

bool book[100000];
bool flag;///判断是否是近亲结婚

bool dfs(int id,int time){
    if(time==5||id==-1||flag==true)
    ///未出五服，id不存在，
        return false;
    book[id]=true;
    if(cp[id].fid!=-1){///如果父亲结点，未访问过
        if(book[cp[id].fid]==false){
            bfs(cp[id].fid,time+1);
        }
        else{///如果父亲结点访问过了
            flag=true;
            return false;
        }
    }
    if(cp[id].mid!=-1){
        if(book[cp[id].mid]==false){///如果母亲结点，未访问过
            bfs(cp[id].mid,time+1);
        }
        else{///如果母亲结点访问过了，
            flag=true;
            return false;
        }
    }
    if(flag==true)  return false;///均未访问过
    return true;
}

int main(){
    int n,m,id,ch1,ch2;
    scanf("%d",&n);
    for(int i=0;i<100000;i++){///需要初始化父母结点，否则无法输出正确结果
        cp[i].fid=-1;
        cp[i].mid=-1;
    }
    for(int i=1;i<=n;i++){
        scanf("%d",&id);
        scanf("%s %d %d",&cp[id].sex,&cp[id].fid,&cp[id].mid);
        cp[cp[id].fid].sex='M';///记录父母的性别
        cp[cp[id].mid].sex='F';
    }
    scanf("%d",&m);
    while(m--){
        scanf("%d%d",&ch1,&ch2);
        memset(book,false,sizeof(book));
        flag=false;
        if(cp[ch1].sex==cp[ch2].sex)///如果性别相同，则不考虑
            printf("Never Mind\n");
        else{
            if(dfs(ch1,0)&&dfs(ch2,0))
                printf("Yes\n");
            else
                printf("No\n");
        }
    }

    return 0;
}
```
