```
#include <bits/stdc++.h>

using namespace std;

const int maxn=10010;

int father[maxn];
int root[maxn],book[maxn];
int n;

int findfather(int x){
    int a=x;
    while(x!=father[x]){
        x=father[x];
    }

    while (a != father[a]) {
		int z = a;
		a = father[a];
		father[z] = x;
	}
    return x;
}

void Union(int a,int b){
    int faA=findfather(a);
    int faB=findfather(b);
    //printf("%d %d;%d %d\n",a,b,faA,faB);
    if(faA>=faB){
        father[faA]=faB;
    }
    else{
        father[faB]=faA;
    }
}

int main(){
    scanf("%d",&n);
    int k,fa,tmp;
    memset(book,0,sizeof(book));
    memset(root,0,sizeof(root));
    for(int i=0;i<n;i++){
        scanf("%d%d",&k,&fa);
        book[tmp]=1;
        for(int j=1;j<k;j++){
            scanf("%d",&tmp);
            Union(fa,tmp);
            book[tmp]=1;
        }
    }
    int tol_p=0;
    for(int i=0;i<maxn;i++){
        if(book[i]) tol_p++;///记录总人数
        root[findfather(i)]++;///记录每个部落人数
    }
    int b_num=0;///计算部落数
    for(int i=0;i<maxn;i++){
        if(root[findfather(i)])
            b_num++;
    }
    printf("%d %d\n",tol_p,b_num);
    int q,a,b;
    scanf("%d",&q);
    for(int i=0;i<q;i++){
        scanf("%d%d",&a,&b);
        if(findfather(a)==findfather(b))
            printf("Y\n");
        else
            printf("N\n");
    }

    return 0;
}
```
