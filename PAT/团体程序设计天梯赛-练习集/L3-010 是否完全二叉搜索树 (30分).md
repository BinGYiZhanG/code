
* ```f1```记录左子树是否存在，如果为```false```,则置```f2```为```true```,也就是左子树存在，右子树不存在

```
#include <bits/stdc++.h>

using namespace std;

const int maxn = 1010;
struct node {
	int data;
	node* lchild;
	node* rchild;
///	int layer;
};
node* newNode(int val) {
	node* Node = new node;
	Node->data = val;
	Node->lchild = Node->rchild = NULL;
	return Node;
}


void insert_(node* &root, int x) {
	if (root == NULL) {
		root = newNode(x);
		return;
	}
	if (x == root->data) {
		insert_(root->lchild, x);
	}
	else if (x < root->data) {
		insert_(root->lchild, x);
	}
	else {
		insert_(root->rchild, x);
	}
}

int n, x;
bool f1=0,f2=0;
void bfs(node* root){
    int num=0;
    queue<node*> que;
    que.push(root);

    while(!que.empty()){
        node* top=que.front();
        que.pop();

        printf("%d",top->data);
        if(num<n-1)
            printf(" ");
        num++;
        if(top->rchild!=NULL){
            if(f1==1)   f2=1;
            que.push(top->rchild);
        }
        else
            f1=1;
        if(top->lchild!=NULL){
            if(f1==1)   f2=1;
            que.push(top->lchild);
        }
        else
            f1=1;
    }
}

int main() {

	node* root = NULL;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &x);
		insert_(root, x);
	}
    bfs(root);
    if(f2==0)
        printf("\nYES\n");
    else
        printf("\nNO\n");

    return 0;
}
```
