void twoPointers(int n,int m){
    int i=0,j=n-1;
    while(i<j){
        if(a[i]+a[j]==m)    break;
        else if(a[i]+a[j]<m)
            i++;
        else
            j--;
    }
    if(i<j)///有解
        printf("%d %d\n",a[i],a[j]);

}
