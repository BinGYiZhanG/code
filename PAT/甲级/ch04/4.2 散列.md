### 1048 Find Coins (25分)
#### 输出

```
V1+V2=M,且V1<V2,如果输出不唯一，输出V1最小的那一对

```

*
```
const int maxn = 1005;
int HashTable[maxn];
using namespace std;
int main() {
	int n, m,a;
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> a;
		++HashTable[a];
	}
	for (int i = 0; i < maxn; i++) {
		if (HashTable[i] && HashTable[m - i]) {
			if (i == m - i && HashTable[i] <= 1) {///如果两个数字相等，且该数字只出现过一次，则跳过
				continue;
			}
			printf("%d %d\n", i, m - i);
			return 0;
		}
	}
	printf("No Solution\n");
	return 0;
}
```
