### 问题 A: 求第k大数

```
#include<bits/stdc++.h>

using namespace std;

priority_queue<int> pq;

int main(){
    int n,k,x;
    while(scanf("%d%d",&n,&k)!=EOF){
        while(n--){
            scanf("%d",&x);
            pq.push(x);
        }
        while(!pq.empty()&&--k){
            //cout<<pq.top()<<endl;
            pq.pop();
        }
        cout<<pq.top()<<endl;
    }
    return 0;
}

```
