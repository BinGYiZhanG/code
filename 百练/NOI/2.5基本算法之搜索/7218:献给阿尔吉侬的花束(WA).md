#include<bits/stdc++.h>

using namespace std;

struct Node{
    int x,y;
};

char mp[210][210];
bool vis[210][210];
int dis[210][210];
int stx,sty,edx,edy;
int R,C,cnt;
int movh[4][2]={{1,0},{0,-1},{-1,0},{0,1}};

void DFS(int x,int y){
    //cnt++;
    vis[x][y]=true;
    if(x==edx&&y==edy){
        return ;
    }
    for(int i=0;i<4;i++){
        int dx=x+movh[i][0];
        int dy=y+movh[i][1];
        if(dx>=R||dx<0||dy>=C||dy<0||mp[dx][dy]=='#'||vis[dx][dy])
            continue;
        dis[dx][dy]=dis[x][y]+1;
        DFS(dx,dy);
    }
}

void BFS(){
    queue<Node> q;
    Node nst;
    nst.x=stx,nst.y=sty;
    q.push(nst);

    while(!q.empty()){
        Node nd=q.front();
        q.pop();
        for(int i=0;i<4;i++){
            int dx=nd.x+movh[i][0];
            int dy=nd.y+movh[i][1];
            if(dx==edx&&dx==edy)
                return ;
            if(vis[dx][dy]||dx>=R||dx<0||dy>=C||dy<0||mp[dx][dy]=='#'||mp[dx][dy]=='S'||mp[dx][dy]=='E')
                continue;
            dis[dx][dy]=dis[nd.x][nd.y]+1;

            Node tmp;
            tmp.x=dx,tmp.y=dy;
            q.push(tmp);
        }
    }
}

int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        scanf("%d%d",&R,&C);
        for(int i=0;i<R;i++){
            scanf("%s",mp[i]);
            for(int j=0;j<C;j++){
                if(mp[i][j]=='S'){
                    stx=i;
                    sty=j;
                }
                if(mp[i][j]=='E'){
                    edx=i;
                    edy=j;
                }
            }
        }
        memset(vis,false,sizeof(vis));
        memset(dis,0,sizeof(dis));
        cnt=0;
        BFS();
        if(dis[edx][edy]==0)
            printf("oop!\n");
        else
            printf("%d\n",dis[edx][edy]);
    }

    return 0;
}
