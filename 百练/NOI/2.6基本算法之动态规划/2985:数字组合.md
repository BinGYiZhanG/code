#include <bits/stdc++.h>

using namespace std;

int dp[1005];
int num[1005];

int main(){
    int n,goal;
    scanf("%d%d",&n,&goal);
    for(int i=1;i<=n;i++)
        scanf("%d",&num[i]);
    dp[0]=1;///0只有一种组合方式
    for(int i=1;i<=n;i++){
        for(int v=goal;v>=num[i];v--){
            dp[v]+=dp[v-num[i]];
            ///一维背包问题：dp[v]=max(dp[v],dp[v-num[i]]+c[i]);
        }
    }

    printf("%d\n",dp[goal]);

    return 0;
}
